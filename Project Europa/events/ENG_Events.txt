	add_namespace = ENG_event

	# The Lollard Heresy
country_event = {
	id = ENG_event.1
	title = ENG_event.1.t
	desc = ENG_event.1.d
	picture = COA_ENG_EP


	fire_only_once = yes
	

	trigger = {
		tag = ENG
		date < 1500.01.01
		num_of_controlled_states > 5
		has_government = neutrality
	}
	
	mean_time_to_happen = {
		days = 1825
	}

	immediate = {
		hidden_effect = {
			set_global_flag = lollard_heresy
			random_owned_state = {
				limit = {
					OR = {
						region = 1
						region = 2
					}
				}
				set_state_flag = lollards
			}
			random_owned_state = {
				limit = {
					OR = {
						region = 1
						region = 2
					}
					NOT = { has_state_flag = lollards }
				}
				set_state_flag = lollards
			}
			random_owned_state = {
				limit = {
					OR = {
						region = 1
						region = 2
					}
					NOT = { has_state_flag = lollards }
				}
				set_state_flag = lollards
			}
			random_owned_state = {
				limit = {
					OR = {
						region = 1
						region = 2
					}
					NOT = { has_state_flag = lollards }
				}
				set_state_flag = lollards
			}
			random_owned_state = {
				limit = {
					OR = {
						region = 1
						region = 2
					}
					NOT = { has_state_flag = lollards }
				}
				set_state_flag = lollards
			}
			random_owned_state = {
				limit = {
					OR = {
						region = 1
						region = 2
					}
					NOT = { has_state_flag = lollards }
				}
				set_state_flag = lollards
			}									
		}
	}

	option = {		# Stop the Heretics
		name = "ENG_event.1.a"
		ai_chance = { factor = 75 }
		every_owned_state  = {
			limit = { has_state_flag = lollards }
			add_state_modifier = {
				modifier ={
					local_resources = -0.5
				}
			}
		}
		
		add_ideas = opposing_lollards
		
		add_opinion_modifier = { target = PAP modifier = opposing_lollards }		
	}
	option = {		# Let them be
		name = "ENG_event.1.b"
		ai_chance = { factor = 25 }
		add_ideas = tolerates_lollards
		add_opinion_modifier = { target = PAP modifier = PAP_tolerates_lollards }
		if = {
			limit = {
				FRA = {
					exists = yes
					has_government = neutrality
				}
			}
			add_opinion_modifier = { target = FRA modifier = tolerates_lollards }
		}
		if = {
			limit = {
				BUR = {
					exists = yes
					has_government = neutrality
				}
			}
			add_opinion_modifier = { target = BUR modifier = tolerates_lollards }
		}
		if = {
			limit = {
				BRI = {
					exists = yes
					has_government = neutrality
				}
			}
			add_opinion_modifier = { target = BRI modifier = tolerates_lollards }
		}
		hidden_effect = {
			every_owned_state  = {
				limit = { has_state_flag = lollards }
				clr_state_flag = lollards
			}
		}
	}
